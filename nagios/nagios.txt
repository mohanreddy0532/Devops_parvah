# Install nagios Server
http://tecadmin.net/install-nagios-core-service-on-centos-rhel/#
STEP1: INSTALL REQUIRED PACKAGES
yum install httpd php php-cli gcc glibc glibc-common gd gd-devel net-snmp
service httpd start

STEP2:
useradd nagios
passwd nagios

groupadd nagcmd
usermod -a -G nagcmd nagios
usermod -a -G nagcmd apache

STEP3******Core Installation
cd /opt/;
wget http://prdownloads.sourceforge.net/sourceforge/nagios/nagios-4.2.4.tar.gz;
tar xzf nagios-4.2.4.tar.gz;
cd nagios-4.2.4;
./configure --with-command-group=nagcmd;
make all;
make install;
make install-init;
make install-config;
make install-commandmode;

make install-webconf

STEP4: CONFIGURE APACHE
htpasswd -c /usr/local/nagios/etc/htpasswd.users nagiosadmin
nagios
service httpd restart
Goto server URL

STEP5: Install Nagios Plugins
cd /opt;
wget http://nagios-plugins.org/download/nagios-plugins-2.1.4.tar.gz;
tar xzf nagios-plugins-2.1.4.tar.gz;
cd nagios-plugins-2.1.4;

./configure --with-nagios-user=nagios --with-nagios-group=nagios
make
make install

STEP6: Verify and Start Nagios
/usr/local/nagios/bin/nagios -v /usr/local/nagios/etc/nagios.cfg
service nagios start

chkconfig --add nagios
chkconfig nagios on

-------------------------------- Install nagios Client:---------------------------

enable epel repo
sudo su -
rpm -ivh http://dl.fedoraproject.org/pub/epel/7/x86_64/e/epel-release-7-9.noarch.rpm

yum install nrpe -y
vi /etc/nagios/nrpe.cfg

allowed_hosts=127.0.0.1
to
